# 학원 사전학습 강의

~~~java
// 반목문 안에 반목문 == 중첩 반복문
// 하위 반복문의 변수명은 상위와 다르게 해줘야함

public class Test10 {

	public static void main(String[] args) {
		System.out.println("구구단을 외우자");
		for(int i=2; i<=9; i++) {
			System.out.println("** "+i+"단 **");
			for(int j=1; j<=9; j++) {
			System.out.println(i+" X "+j+" = "+i*j);
			}
		}
		System.out.println("=============");
		System.out.println("이번주에 모두 외우자!!");-*-
	}
}
~~~
