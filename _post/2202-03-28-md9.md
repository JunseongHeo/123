import java.util.Scanner;

		int amount = scanner.nextInt();
		int price = 1000;
		String memo="(정상가)";
		if(amount>=10) {
			price=800;
			memo="(할인가적용)";
		}
		System.out.println("판매금액은 "+memo+amount*price+"원 입니다.");

// 다양한 연산자
